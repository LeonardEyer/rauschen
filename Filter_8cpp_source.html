40 void Filter::block(q15_t *pSrc, q15_t *pRef, q15_t *pOut, q15_t *pErr, uint32_t blockSize)
41 {
42  q15_t *pState = buffer; /* State pointer */
43  q15_t *pCoeffs = coefficients; /* Coefficient pointer */
44  q15_t *pStateCurnt; /* Points to the current sample of the state */
45  q15_t *px, *pb; /* Temporary pointers for state and coefficient buffers */
46  uint32_t tapCnt, blkCnt; /* Loop counters */
47  q63_t acc; /* Accumulator */
48  q15_t e = 0, d = 0; /* Error, reference data sample */
49  q15_t w = 0, in; /* Weight factor and state */
50  q15_t errorXmu, oneByEnergy; /* Temporary variables to store error and mu product and reciprocal of energy */
51  q15_t postShift; /* Post shift to be applied to weight after reciprocal calculation */
52  q31_t coef; /* Temporary variable for coefficient */
53  q31_t acc_l, acc_h; /* Temporary input */
54  int32_t lShift = (15 - 0); /* Post shift */
55  int32_t uShift = (32 - lShift);
56 
57  /* S->pState points to buffer which contains previous frame (numTaps - 1) samples */
58  /* pStateCurnt points to the location where the new input data should be written */
59  pStateCurnt = &(buffer[(numTaps - 1U)]);
60 
61  /* initialise loop count */
62  blkCnt = blockSize;
63 
64  while (blkCnt > 0U)
65  {
66  /* Copy the new input sample into the state buffer */
67  *pStateCurnt++ = *pSrc;
68 
69  /* Initialize pState pointer */
70  px = pState;
71 
72  /* Initialize coefficient pointer */
73  pb = pCoeffs;
74 
75  /* Read the sample from input buffer */
76  in = *pSrc++;
77 
78  /* Update the energy calculation */
79  energy -= (((q31_t)x0 * (x0)) >> 15);
80  energy += (((q31_t)in * (in)) >> 15);
81 
82  /* Set the accumulator to zero */
83  acc = 0;
84 
85  /* Initialize tapCnt with number of samples */
86  tapCnt = numTaps;
87 
88  while (tapCnt > 0U)
89  {
90  /* Perform the multiply-accumulate */
91  acc += (q63_t)(((q31_t)(*px++) * (*pb++)));
92 
93  /* Decrement the loop counter */
94  tapCnt--;
95  }
96 
97  /* Calc lower part of acc */
98  acc_l = acc & 0xffffffff;
99 
100  /* Calc upper part of acc */
101  acc_h = (acc >> 32) & 0xffffffff;
102 
103  /* Apply shift for lower part of acc and upper part of acc */
104  acc = (uint32_t)acc_l >> lShift | acc_h << uShift;
105 
106  /* Converting the result to 1.15 format and saturate the output */
107  acc = __SSAT(acc, 16U);
108 
109  /* Store the result from accumulator into the destination buffer. */
110  *pOut++ = (q15_t)acc;
111 
112  /* Compute and store error */
113  d = *pRef++;
114  e = d - (q15_t)acc;
115  *pErr++ = e;
116 
117  /* Calculation of 1/energy */
118  postShift = arm_recip_q15((q15_t)energy + DELTA_Q15, &oneByEnergy, recipTable);
119 
120  /* Calculation of e * mu value */
121  errorXmu = (q15_t)(((q31_t)e * mu) >> 15);
122 
123  /* Calculation of (e * mu) * (1/energy) value */
124  acc = (((q31_t)errorXmu * oneByEnergy) >> (15 - postShift));
125 
126  /* Weighting factor for the normalized version */
127  w = (q15_t)__SSAT((q31_t)acc, 16);
128 
129  /* Initialize pState pointer */
130  px = pState;
131 
132  /* Initialize coefficient pointer */
133  pb = pCoeffs;
134 
135  //if(freezeCoeffs == false) {
136  /* Initialize tapCnt with number of samples */
137  tapCnt = numTaps;
138

140  {
141  /* Perform the multiply-accumulate */
142  coef = (q31_t)*pb + (((q31_t)w * (*px++)) >> 15);
143  *pb++ = (q15_t)__SSAT(coef, 16);
144 
145  /* Decrement loop counter */
146  tapCnt--;
147  }
148  //}
149 
150  x0 = *pState;
151 
152  /* Advance state pointer by 1 for the next sample */
153  pState = pState + 1;
154 
155  /* Decrement loop counter */
156  blkCnt--;
157  }
158 
159  /* Processing is complete.
160  Now copy the last numTaps - 1 samples to the start of the state buffer.
161  This prepares the state buffer for the next function call. */
162 
163  /* Points to the start of the pState buffer */
164  pStateCurnt = buffer;
165 
166  /* Initialize tapCnt with number of samples */
167  tapCnt = (numTaps - 1U);
168

170  {
171  *pStateCurnt++ = *pState++;
172 
173  /* Decrement loop counter */
174  tapCnt--;
